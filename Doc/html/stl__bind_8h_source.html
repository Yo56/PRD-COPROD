22 template <typename T> struct container_traits {
23  template <typename T2> static std::true_type test_comparable(decltype(std::declval<const T2 &>() == std::declval<const T2 &>())*);
24  template <typename T2> static std::false_type test_comparable(...);
25  template <typename T2> static std::true_type test_value(typename T2::value_type *);
26  template <typename T2> static std::false_type test_value(...);
27  template <typename T2> static std::true_type test_pair(typename T2::first_type *, typename T2::second_type *);
28  template <typename T2> static std::false_type test_pair(...);
29 
30  static constexpr const bool is_comparable = std::is_same<std::true_type, decltype(test_comparable<T>(nullptr))>::value;
31  static constexpr const bool is_pair = std::is_same<std::true_type, decltype(test_pair<T>(nullptr, nullptr))>::value;
32  static constexpr const bool is_vector = std::is_same<std::true_type, decltype(test_value<T>(nullptr))>::value;
33  static constexpr const bool is_element = !is_pair && !is_vector;

337 // Add the buffer interface to a vector
338 template <typename Vector, typename Class_, typename... Args>
339 enable_if_t<detail::any_of<std::is_same<Args, buffer_protocol>...>::value>
340 vector_buffer(Class_& cl) {
341  using T = typename Vector::value_type;
342 
343  static_assert(vector_has_data_and_format<Vector>::value, "There is not an appropriate format descriptor for this vector");
344 
345  // numpy.h declares this for arbitrary types, but it may raise an exception and crash hard at runtime if PYBIND11_NUMPY_DTYPE hasn't been called, so check here
346  format_descriptor<T>::format();
347 
348  cl.def_buffer([](Vector& v) -> buffer_info {
349  return buffer_info(v.data(), static_cast<ssize_t>(sizeof(T)), format_descriptor<T>::format(), 1, {v.size()}, {sizeof(T)});
350  });
351 
352  cl.def("__init__", [](Vector& vec, buffer buf) {
353  auto info = buf.request();
354  if (info.ndim != 1 || info.strides[0] % static_cast<ssize_t>(sizeof(T)))
355  throw type_error("Only valid 1D buffers can be copied to a vector");
356  if (!detail::compare_buffer_info<T>::compare(info) || (ssize_t) sizeof(T) != info.itemsize)
357  throw type_error("Format mismatch (Python: " + info.format + " C++: " + format_descriptor<T>::format() + ")");
358  new (&vec) Vector();
359  vec.reserve((size_t) info.shape[0]);
360  T *p = static_cast<T*>(info.ptr);
361  ssize_t step = info.strides[0] / static_cast<ssize_t>(sizeof(T));
362  T *end = p + info.shape[0] * step;
363  for (; p != end; p += step)
364  vec.push_back(*p);
365  });
366 
367  return;
368 }
