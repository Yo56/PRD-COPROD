246  PYBIND11_NOINLINE void add_base(const std::type_info &base, void *(*caster)(void *)) {
247  auto base_info = detail::get_type_info(base, false);
248  if (!base_info) {
249  std::string tname(base.name());
250  detail::clean_type_id(tname);
251  pybind11_fail("generic_type: type \"" + std::string(name) +
252  "\" referenced unknown base type \"" + tname + "\"");
253  }
254 
255  if (default_holder != base_info->default_holder) {
256  std::string tname(base.name());
257  detail::clean_type_id(tname);
258  pybind11_fail("generic_type: type \"" + std::string(name) + "\" " +
259  (default_holder ? "does not have" : "has") +
260  " a non-default holder type while its base \"" + tname + "\" " +
261  (base_info->default_holder ? "does not" : "does"));
262  }
263 
264  bases.append((PyObject *) base_info->type);
265 
266  if (base_info->type->tp_dictoffset != 0)
267  dynamic_attr = true;
268 
269  if (caster)
270  base_info->implicit_casts.emplace_back(type, caster);
271  }
