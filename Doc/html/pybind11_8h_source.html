102  /* Store the function including any extra state it might have (e.g. a lambda capture object) */

191  void initialize_generic(detail::function_record *rec, const char *text,
192  const std::type_info *const *types, size_t args) {
193 
194  /* Create copies of all referenced C-style strings */
195  rec->name = strdup(rec->name ? rec->name : "");
196  if (rec->doc) rec->doc = strdup(rec->doc);
197  for (auto &a: rec->args) {
198  if (a.name)
199  a.name = strdup(a.name);
200  if (a.descr)
201  a.descr = strdup(a.descr);
202  else if (a.value)
203  a.descr = strdup(a.value.attr("__repr__")().cast<std::string>().c_str());
204  }
205 
206  /* Generate a proper function signature */
207  std::string signature;
208  size_t type_depth = 0, char_index = 0, type_index = 0, arg_index = 0;
209  while (true) {
210  char c = text[char_index++];
211  if (c == '\0')
212  break;
213 
214  if (c == '{') {
215  // Write arg name for everything except *args, **kwargs and return type.
216  if (type_depth == 0 && text[char_index] != '*' && arg_index < args) {
217  if (!rec->args.empty() && rec->args[arg_index].name) {
218  signature += rec->args[arg_index].name;
219  } else if (arg_index == 0 && rec->is_method) {
220  signature += "self";
221  } else {
222  signature += "arg" + std::to_string(arg_index - (rec->is_method ? 1 : 0));
223  }
224  signature += ": ";
225  }
226  ++type_depth;
227  } else if (c == '}') {
228  --type_depth;
229  if (type_depth == 0) {
230  if (arg_index < rec->args.size() && rec->args[arg_index].descr) {
231  signature += "=";
232  signature += rec->args[arg_index].descr;
233  }
234  arg_index++;
235  }
236  } else if (c == '%') {
237  const std::type_info *t = types[type_index++];
238  if (!t)
239  pybind11_fail("Internal error while parsing type signature (1)");
240  if (auto tinfo = detail::get_type_info(*t)) {
241 #if defined(PYPY_VERSION)
242  signature += handle((PyObject *) tinfo->type)
243  .attr("__module__")
244  .cast<std::string>() + ".";
245 #endif
246  signature += tinfo->type->tp_name;
247  } else {
248  std::string tname(t->name());
249  detail::clean_type_id(tname);
250  signature += tname;
251  }
252  } else {
253  signature += c;
254  }
255  }
256  if (type_depth != 0 || types[type_index] != nullptr)
257  pybind11_fail("Internal error while parsing type signature (2)");
258 
259  #if !defined(PYBIND11_CONSTEXPR_DESCR)
260  delete[] types;
261  delete[] text;
262  #endif
263 
264 #if PY_MAJOR_VERSION < 3
265  if (strcmp(rec->name, "__next__") == 0) {
266  std::free(rec->name);
267  rec->name = strdup("next");
268  } else if (strcmp(rec->name, "__bool__") == 0) {
269  std::free(rec->name);
270  rec->name = strdup("__nonzero__");
271  }
272 #endif
273  rec->signature = strdup(signature.c_str());
274  rec->args.shrink_to_fit();
275  rec->is_constructor = !strcmp(rec->name, "__init__") || !strcmp(rec->name, "__setstate__");
276  rec->nargs = (std::uint16_t) args;
277 
278  if (rec->sibling && PYBIND11_INSTANCE_METHOD_CHECK(rec->sibling.ptr()))
279  rec->sibling = PYBIND11_INSTANCE_METHOD_GET_FUNCTION(rec->sibling.ptr());
280 
281  detail::function_record *chain = nullptr, *chain_start = rec;
282  if (rec->sibling) {
283  if (PyCFunction_Check(rec->sibling.ptr())) {
284  auto rec_capsule = reinterpret_borrow<capsule>(PyCFunction_GET_SELF(rec->sibling.ptr()));
285  chain = (detail::function_record *) rec_capsule;
286  /* Never append a method to an overload chain of a parent class;
287  instead, hide the parent's overloads in this case */
288  if (!chain->scope.is(rec->scope))
289  chain = nullptr;
290  }
291  // Don't trigger for things like the default __init__, which are wrapper_descriptors that we are intentionally replacing
292  else if (!rec->sibling.is_none() && rec->name[0] != '_')
293  pybind11_fail("Cannot overload existing non-function object \"" + std::string(rec->name) +
294  "\" with a function of the same name");
295  }
296 
297  if (!chain) {
298  /* No existing overload was found, create a new function object */
299  rec->def = new PyMethodDef();
300  std::memset(rec->def, 0, sizeof(PyMethodDef));
301  rec->def->ml_name = rec->name;
302  rec->def->ml_meth = reinterpret_cast<PyCFunction>(*dispatcher);
303  rec->def->ml_flags = METH_VARARGS | METH_KEYWORDS;
304 
305  capsule rec_capsule(rec, [](void *ptr) {
306  destruct((detail::function_record *) ptr);
307  });
308 
309  object scope_module;
310  if (rec->scope) {
311  if (hasattr(rec->scope, "__module__")) {
312  scope_module = rec->scope.attr("__module__");
313  } else if (hasattr(rec->scope, "__name__")) {
314  scope_module = rec->scope.attr("__name__");
315  }
316  }
317 
318  m_ptr = PyCFunction_NewEx(rec->def, rec_capsule.ptr(), scope_module.ptr());
319  if (!m_ptr)
320  pybind11_fail("cpp_function::cpp_function(): Could not allocate function object");
321  } else {
322  /* Append at the end of the overload chain */
323  m_ptr = rec->sibling.ptr();
324  inc_ref();
325  chain_start = chain;
326  if (chain->is_method != rec->is_method)
327  pybind11_fail("overloading a method with both static and instance methods is not supported; "
328  #if defined(NDEBUG)
329  "compile in debug mode for more details"
330  #else
331  "error while attempting to bind " + std::string(rec->is_method ? "instance" : "static") + " method " +
332  std::string(pybind11::str(rec->scope.attr("__name__"))) + "." + std::string(rec->name) + signature
333  #endif
334  );

336  chain = chain->next;
337  chain->next = rec;
338  }

341  int index = 0;
342  /* Create a nice pydoc rec including all signatures and
343  docstrings of the functions in the overload chain */
344  if (chain && options::show_function_signatures()) {
345  // First a generic signature
346  signatures += rec->name;
347  signatures += "(*args, **kwargs)\n";
348  signatures += "Overloaded function.\n\n";

356  signatures += std::to_string(++index) + ". ";

358  signatures += it->signature;
359  signatures += "\n";

365  if (first_user_def) first_user_def = false;
366  else signatures += "\n";
367  }
368  if (options::show_function_signatures()) signatures += "\n";
369  signatures += it->doc;
370  if (options::show_function_signatures()) signatures += "\n";
371  }
372  }

376  if (func->m_ml->ml_doc)
377  std::free(const_cast<char *>(func->m_ml->ml_doc));
378  func->m_ml->ml_doc = strdup(signatures.c_str());
379 
380  if (rec->is_method) {
381  m_ptr = PYBIND11_INSTANCE_METHOD_NEW(m_ptr, rec->scope.ptr());
382  if (!m_ptr)
383  pybind11_fail("cpp_function::cpp_function(): Could not allocate instance method object");
384  Py_DECREF(func);
385  }

412  static PyObject *dispatcher(PyObject *self, PyObject *args_in, PyObject *kwargs_in) {
413  using namespace detail;
414 
415  /* Iterator over the list of potentially admissible overloads */
416  function_record *overloads = (function_record *) PyCapsule_GetPointer(self, nullptr),
417  *it = overloads;
418 
419  /* Need to know how many arguments + keyword arguments there are to pick the right overload */
420  const size_t n_args_in = (size_t) PyTuple_GET_SIZE(args_in);
421 
422  handle parent = n_args_in > 0 ? PyTuple_GET_ITEM(args_in, 0) : nullptr,
423  result = PYBIND11_TRY_NEXT_OVERLOAD;
424 
425  try {
426  // We do this in two passes: in the first pass, we load arguments with `convert=false`;
427  // in the second, we allow conversion (except for arguments with an explicit
428  // py::arg().noconvert()). This lets us prefer calls without conversion, with
429  // conversion as a fallback.
430  std::vector<function_call> second_pass;
431 
432  // However, if there are no overloads, we can just skip the no-convert pass entirely
433  const bool overloaded = it != nullptr && it->next != nullptr;
434 
435  for (; it != nullptr; it = it->next) {
436 
437  /* For each overload:
438  1. Copy all positional arguments we were given, also checking to make sure that
439  named positional arguments weren't *also* specified via kwarg.
440  2. If we weren't given enough, try to make up the omitted ones by checking
441  whether they were provided by a kwarg matching the `py::arg("name")` name. If
442  so, use it (and remove it from kwargs; if not, see if the function binding
443  provided a default that we can use.
444  3. Ensure that either all keyword arguments were "consumed", or that the function
445  takes a kwargs argument to accept unconsumed kwargs.
446  4. Any positional arguments still left get put into a tuple (for args), and any
447  leftover kwargs get put into a dict.
448  5. Pack everything into a vector; if we have py::args or py::kwargs, they are an
449  extra tuple or dict at the end of the positional arguments.
450  6. Call the function call dispatcher (function_record::impl)
451 
452  If one of these fail, move on to the next overload and keep trying until we get a
453  result other than PYBIND11_TRY_NEXT_OVERLOAD.
454  */
455 
456  function_record &func = *it;
457  size_t pos_args = func.nargs; // Number of positional arguments that we need
458  if (func.has_args) --pos_args; // (but don't count py::args
459  if (func.has_kwargs) --pos_args; // or py::kwargs)
460 
461  if (!func.has_args && n_args_in > pos_args)
462  continue; // Too many arguments for this overload
463 
464  if (n_args_in < pos_args && func.args.size() < pos_args)
465  continue; // Not enough arguments given, and not enough defaults to fill in the blanks
466 
467  function_call call(func, parent);
468 
469  size_t args_to_copy = std::min(pos_args, n_args_in);
470  size_t args_copied = 0;
471 
472  // 1. Copy any position arguments given.
473  bool bad_arg = false;
474  for (; args_copied < args_to_copy; ++args_copied) {
475  argument_record *arg_rec = args_copied < func.args.size() ? &func.args[args_copied] : nullptr;
476  if (kwargs_in && arg_rec && arg_rec->name && PyDict_GetItemString(kwargs_in, arg_rec->name)) {
477  bad_arg = true;
478  break;
479  }
480 
481  handle arg(PyTuple_GET_ITEM(args_in, args_copied));
482  if (arg_rec && !arg_rec->none && arg.is_none()) {
483  bad_arg = true;
484  break;
485  }
486  call.args.push_back(arg);
487  call.args_convert.push_back(arg_rec ? arg_rec->convert : true);
488  }
489  if (bad_arg)
490  continue; // Maybe it was meant for another overload (issue #688)
491 
492  // We'll need to copy this if we steal some kwargs for defaults
493  dict kwargs = reinterpret_borrow<dict>(kwargs_in);
494 
495  // 2. Check kwargs and, failing that, defaults that may help complete the list
496  if (args_copied < pos_args) {
497  bool copied_kwargs = false;
498 
499  for (; args_copied < pos_args; ++args_copied) {
500  const auto &arg = func.args[args_copied];
501 
502  handle value;
503  if (kwargs_in && arg.name)
504  value = PyDict_GetItemString(kwargs.ptr(), arg.name);
505 
506  if (value) {
507  // Consume a kwargs value
508  if (!copied_kwargs) {
509  kwargs = reinterpret_steal<dict>(PyDict_Copy(kwargs.ptr()));
510  copied_kwargs = true;
511  }
512  PyDict_DelItemString(kwargs.ptr(), arg.name);
513  } else if (arg.value) {
514  value = arg.value;
515  }
516 
517  if (value) {
518  call.args.push_back(value);
519  call.args_convert.push_back(arg.convert);
520  }
521  else
522  break;
523  }
524 
525  if (args_copied < pos_args)
526  continue; // Not enough arguments, defaults, or kwargs to fill the positional arguments
527  }
528 
529  // 3. Check everything was consumed (unless we have a kwargs arg)
530  if (kwargs && kwargs.size() > 0 && !func.has_kwargs)
531  continue; // Unconsumed kwargs, but no py::kwargs argument to accept them
532 
533  // 4a. If we have a py::args argument, create a new tuple with leftovers
534  tuple extra_args;
535  if (func.has_args) {
536  if (args_to_copy == 0) {
537  // We didn't copy out any position arguments from the args_in tuple, so we
538  // can reuse it directly without copying:
539  extra_args = reinterpret_borrow<tuple>(args_in);
540  } else if (args_copied >= n_args_in) {
541  extra_args = tuple(0);
542  } else {
543  size_t args_size = n_args_in - args_copied;
544  extra_args = tuple(args_size);
545  for (size_t i = 0; i < args_size; ++i) {
546  handle item = PyTuple_GET_ITEM(args_in, args_copied + i);
547  extra_args[i] = item.inc_ref().ptr();
548  }
549  }
550  call.args.push_back(extra_args);
551  call.args_convert.push_back(false);
552  }
553 
554  // 4b. If we have a py::kwargs, pass on any remaining kwargs
555  if (func.has_kwargs) {
556  if (!kwargs.ptr())
557  kwargs = dict(); // If we didn't get one, send an empty one
558  call.args.push_back(kwargs);
559  call.args_convert.push_back(false);
560  }
561 
562  // 5. Put everything in a vector. Not technically step 5, we've been building it
563  // in `call.args` all along.
564  #if !defined(NDEBUG)
565  if (call.args.size() != func.nargs || call.args_convert.size() != func.nargs)
566  pybind11_fail("Internal error: function call dispatcher inserted wrong number of arguments!");
567  #endif
568 
569  std::vector<bool> second_pass_convert;
570  if (overloaded) {
571  // We're in the first no-convert pass, so swap out the conversion flags for a
572  // set of all-false flags. If the call fails, we'll swap the flags back in for
573  // the conversion-allowed call below.
574  second_pass_convert.resize(func.nargs, false);
575  call.args_convert.swap(second_pass_convert);
576  }
577 
578  // 6. Call the function.
579  try {
580  loader_life_support guard{};
581  result = func.impl(call);
582  } catch (reference_cast_error &) {
583  result = PYBIND11_TRY_NEXT_OVERLOAD;
584  }
585 
586  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD)
587  break;
588 
589  if (overloaded) {
590  // The (overloaded) call failed; if the call has at least one argument that
591  // permits conversion (i.e. it hasn't been explicitly specified `.noconvert()`)
592  // then add this call to the list of second pass overloads to try.
593  for (size_t i = func.is_method ? 1 : 0; i < pos_args; i++) {
594  if (second_pass_convert[i]) {
595  // Found one: swap the converting flags back in and store the call for
596  // the second pass.
597  call.args_convert.swap(second_pass_convert);
598  second_pass.push_back(std::move(call));
599  break;
600  }
601  }
602  }
603  }
604 
605  if (overloaded && !second_pass.empty() && result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
606  // The no-conversion pass finished without success, try again with conversion allowed
607  for (auto &call : second_pass) {
608  try {
609  loader_life_support guard{};
610  result = call.func.impl(call);
611  } catch (reference_cast_error &) {
612  result = PYBIND11_TRY_NEXT_OVERLOAD;
613  }
614 
615  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD)
616  break;
617  }
618  }
619  } catch (error_already_set &e) {
620  e.restore();
621  return nullptr;
622  } catch (...) {
623  /* When an exception is caught, give each registered exception
624  translator a chance to translate it to a Python exception
625  in reverse order of registration.
626 
627  A translator may choose to do one of the following:
628 
629  - catch the exception and call PyErr_SetString or PyErr_SetObject
630  to set a standard (or custom) Python exception, or
631  - do nothing and let the exception fall through to the next translator, or
632  - delegate translation to the next translator by throwing a new type of exception. */
633 
634  auto last_exception = std::current_exception();
635  auto &registered_exception_translators = get_internals().registered_exception_translators;
636  for (auto& translator : registered_exception_translators) {
637  try {
638  translator(last_exception);
639  } catch (...) {
640  last_exception = std::current_exception();
641  continue;
642  }
643  return nullptr;
644  }
645  PyErr_SetString(PyExc_SystemError, "Exception escaped from default exception translator!");
646  return nullptr;
647  }
648 
649  if (result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
650  if (overloads->is_operator)
651  return handle(Py_NotImplemented).inc_ref().ptr();
652 
653  std::string msg = std::string(overloads->name) + "(): incompatible " +
654  std::string(overloads->is_constructor ? "constructor" : "function") +
655  " arguments. The following argument types are supported:\n";
656 
657  int ctr = 0;
658  for (function_record *it2 = overloads; it2 != nullptr; it2 = it2->next) {
659  msg += " "+ std::to_string(++ctr) + ". ";
660 
661  bool wrote_sig = false;
662  if (overloads->is_constructor) {
663  // For a constructor, rewrite `(self: Object, arg0, ...) -> NoneType` as `Object(arg0, ...)`
664  std::string sig = it2->signature;
665  size_t start = sig.find('(') + 7; // skip "(self: "
666  if (start < sig.size()) {
667  // End at the , for the next argument
668  size_t end = sig.find(", "), next = end + 2;
669  size_t ret = sig.rfind(" -> ");
670  // Or the ), if there is no comma:
671  if (end >= sig.size()) next = end = sig.find(')');
672  if (start < end && next < sig.size()) {
673  msg.append(sig, start, end - start);
674  msg += '(';
675  msg.append(sig, next, ret - next);
676  wrote_sig = true;
677  }
678  }
679  }
680  if (!wrote_sig) msg += it2->signature;
681 
682  msg += "\n";
683  }
684  msg += "\nInvoked with: ";
685  auto args_ = reinterpret_borrow<tuple>(args_in);
686  bool some_args = false;
687  for (size_t ti = overloads->is_constructor ? 1 : 0; ti < args_.size(); ++ti) {
688  if (!some_args) some_args = true;
689  else msg += ", ";
690  msg += pybind11::repr(args_[ti]);
691  }
692  if (kwargs_in) {
693  auto kwargs = reinterpret_borrow<dict>(kwargs_in);
694  if (kwargs.size() > 0) {
695  if (some_args) msg += "; ";
696  msg += "kwargs: ";
697  bool first = true;
698  for (auto kwarg : kwargs) {
699  if (first) first = false;
700  else msg += ", ";
701  msg += pybind11::str("{}={!r}").format(kwarg.first, kwarg.second);
702  }
703  }
704  }
705 
706  PyErr_SetString(PyExc_TypeError, msg.c_str());
707  return nullptr;
708  } else if (!result) {
709  std::string msg = "Unable to convert function return value to a "
710  "Python type! The signature was\n\t";
711  msg += it->signature;
712  PyErr_SetString(PyExc_TypeError, msg.c_str());
713  return nullptr;
714  } else {
715  if (overloads->is_constructor) {
716  auto tinfo = get_type_info((PyTypeObject *) overloads->scope.ptr());
717  tinfo->init_holder(reinterpret_cast<instance *>(parent.ptr()), nullptr);
718  }
719  return result.ptr();
720  }
721  }

1356  if (!nurse || !patient)
1357  pybind11_fail("Could not activate keep_alive!");
1358 
1359  if (patient.is_none() || nurse.is_none())
1360  return; /* Nothing to keep alive or nothing to be kept alive by */
