175  template<typename T, typename C>
176  std::size_t find_or_add(C &container, const T &item)
177  {
178 #pragma GCC diagnostic push
179 #pragma GCC diagnostic ignored "-Wsign-conversion"
180  auto iter = std::find(container.begin(), container.end(), item);
181  if (iter != container.end())
182  {
183  return std::size_t(iter - container.begin());
184  }
185  iter = container.emplace(container.end(), item);
186  return std::size_t(iter - container.begin());
187 #pragma GCC diagnostic pop
188  }

190  template<typename T>
191  std::unordered_map<std::size_t, std::size_t> garbage_collect(
192  const std::unordered_map<std::size_t, std::size_t> &reference_counts,
193  std::vector<T> &container)
194  {
195  std::unordered_map<std::size_t, std::size_t> id_map;
196  std::size_t unreferenced = 0;
197  const auto original_size = container.size();
198 
199  for (std::size_t i = 0; i < original_size; ++i)
200  {
201  id_map[i] = i - unreferenced;
202 
203  if (reference_counts.count(i) == 0 || reference_counts.at(i) == 0)
204  {
205  container.erase(container.begin() + static_cast<typename std::vector<T>::difference_type>(i - unreferenced));
206  unreferenced++;
207  }
208  }
209 
210  return id_map;
211  }

379  format_impl *find_or_create(format_impl &pattern)
380  {
381  pattern.references = 0;
382  std::size_t id = 0;
383  auto iter = format_impls.begin();
384  while (iter != format_impls.end() && !(*iter == pattern))
385  {
386  ++id;
387  ++iter;
388  }
389  if (iter == format_impls.end())
390  {
391  iter = format_impls.emplace(format_impls.end(), pattern);
392  }
393  auto &result = *iter;
394 
395  result.parent = this;
396  result.id = id;
397  result.references++;
398 
399  if (id != pattern.id)
400  {
401  iter = format_impls.begin();
402  std::advance(iter, static_cast<std::list<format_impl>::difference_type>(pattern.id));
403  iter->references -= iter->references > 0 ? 1 : 0;
404  garbage_collect();
405  }
406 
407  return &result;
408  }
