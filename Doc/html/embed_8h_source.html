142 inline void finalize_interpreter() {
143  handle builtins(PyEval_GetBuiltins());
144  const char *id = PYBIND11_INTERNALS_ID;
145 
146  // Get the internals pointer (without creating it if it doesn't exist). It's possible for the
147  // internals to be created during Py_Finalize() (e.g. if a py::capsule calls `get_internals()`
148  // during destruction), so we get the pointer-pointer here and check it after Py_Finalize().
149  detail::internals **internals_ptr_ptr = &detail::get_internals_ptr();
150  // It could also be stashed in builtins, so look there too:
151  if (builtins.contains(id) && isinstance<capsule>(builtins[id]))
152  internals_ptr_ptr = capsule(builtins[id]);
153 
154  Py_Finalize();
155 
156  if (internals_ptr_ptr) {
157  delete *internals_ptr_ptr;
158  *internals_ptr_ptr = nullptr;
159  }
160 }
