173 template<>
174 pattern_fill_type from_string(const std::string &string)
175 {
176 #pragma clang diagnostic push
177 #pragma clang diagnostic ignored "-Wexit-time-destructors"
178  static std::unordered_map<std::string, pattern_fill_type> patternFill {
179  {"darkdown", pattern_fill_type::darkdown },
180  { "darkgray", pattern_fill_type::darkgray },
181  { "darkgrid", pattern_fill_type::darkgrid },
182  { "darkhorizontal", pattern_fill_type::darkhorizontal },
183  { "darktrellis", pattern_fill_type::darktrellis },
184  { "darkup", pattern_fill_type::darkup },
185  { "darkvertical", pattern_fill_type::darkvertical },
186  { "gray0625", pattern_fill_type::gray0625 },
187  { "gray125", pattern_fill_type::gray125 },
188  { "lightdown", pattern_fill_type::lightdown },
189  { "lightgray", pattern_fill_type::lightgray },
190  { "lightgrid", pattern_fill_type::lightgrid },
191  { "lighthorizontal", pattern_fill_type::lighthorizontal },
192  { "lighttrellis", pattern_fill_type::lighttrellis },
193  { "lightup", pattern_fill_type::lightup },
194  { "lightvertical", pattern_fill_type::lightvertical },
195  { "mediumgray", pattern_fill_type::mediumgray },
196  { "none", pattern_fill_type::none },
197  { "solid", pattern_fill_type::solid }
198  };
199 #pragma clang diagnostic pop
200 
201  auto toLower = [](std::string str) {
202  auto bg{ std::begin (str) };
203  auto en{ std::end (str) };
204  std::transform(bg, en, bg,
205  [](char c) {
206  // static cast to avoid int -> char narrowing warning
207  return static_cast<char>(tolower(c));
208  });
209 
210  return str;
211  };
212 
213  auto patternLookup = [](const std::string& key) {
214  auto entry { patternFill.find (key) };
215  if (entry != std::end (patternFill)) {
216  return entry->second;
217  }
218  else {
219  // Note: there won't be an error if there is an unsupported pattern
220  return pattern_fill_type::none;
221  }
222  };
223 
224  std::string lowerString {toLower (string) };
225 
226  return patternLookup (lowerString);
227 }
