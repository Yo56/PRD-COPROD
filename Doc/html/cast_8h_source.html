167 PYBIND11_NOINLINE inline void all_type_info_populate(PyTypeObject *t, std::vector<type_info *> &bases) {
168  std::vector<PyTypeObject *> check;
169  for (handle parent : reinterpret_borrow<tuple>(t->tp_bases))
170  check.push_back((PyTypeObject *) parent.ptr());
171 
172  auto const &type_dict = get_internals().registered_types_py;
173  for (size_t i = 0; i < check.size(); i++) {
174  auto type = check[i];
175  // Ignore Python2 old-style class super type:
176  if (!PyType_Check((PyObject *) type)) continue;
177 
178  // Check `type` in the current set of registered python types:
179  auto it = type_dict.find(type);
180  if (it != type_dict.end()) {
181  // We found a cache entry for it, so it's either pybind-registered or has pre-computed
182  // pybind bases, but we have to make sure we haven't already seen the type(s) before: we
183  // want to follow Python/virtual C++ rules that there should only be one instance of a
184  // common base.
185  for (auto *tinfo : it->second) {
186  // NB: Could use a second set here, rather than doing a linear search, but since
187  // having a large number of immediate pybind11-registered types seems fairly
188  // unlikely, that probably isn't worthwhile.
189  bool found = false;
190  for (auto *known : bases) {
191  if (known == tinfo) { found = true; break; }
192  }
193  if (!found) bases.push_back(tinfo);
194  }
195  }
196  else if (type->tp_bases) {
197  // It's some python type, so keep follow its bases classes to look for one or more
198  // registered types
199  if (i + 1 == check.size()) {
200  // When we're at the end, we can pop off the current element to avoid growing
201  // `check` when adding just one base (which is typical--.e. when there is no
202  // multiple inheritance)
203  check.pop_back();
204  i--;
205  }
206  for (handle parent : reinterpret_borrow<tuple>(type->tp_bases))
207  check.push_back((PyTypeObject *) parent.ptr());
208  }
209  }
210 }

383 PYBIND11_NOINLINE inline void instance::allocate_layout() {
384  auto &tinfo = all_type_info(Py_TYPE(this));
385 
386  const size_t n_types = tinfo.size();
387 
388  if (n_types == 0)
389  pybind11_fail("instance allocation failed: new instance has no pybind11-registered base types");
390 
391  simple_layout =
392  n_types == 1 && tinfo.front()->holder_size_in_ptrs <= instance_simple_holder_in_ptrs();
393 
394  // Simple path: no python-side multiple inheritance, and a small-enough holder
395  if (simple_layout) {
396  simple_value_holder[0] = nullptr;
397  simple_holder_constructed = false;
398  }
399  else { // multiple base types or a too-large holder
400  // Allocate space to hold: [v1*][h1][v2*][h2]...[bb...] where [vN*] is a value pointer,
401  // [hN] is the (uninitialized) holder instance for value N, and [bb...] is a set of bool
402  // values that tracks whether each associated holder has been initialized. Each [block] is
403  // padded, if necessary, to an integer multiple of sizeof(void *).
404  size_t space = 0;
405  for (auto t : tinfo) {
406  space += 1; // value pointer
407  space += t->holder_size_in_ptrs; // holder instance
408  }
409  size_t flags_at = space;
410  space += size_in_ptrs(n_types * sizeof(bool)); // holder constructed flags
411 
412  // Allocate space for flags, values, and holders, and initialize it to 0 (flags and values,
413  // in particular, need to be 0). Use Python's memory allocation functions: in Python 3.6
414  // they default to using pymalloc, which is designed to be efficient for small allocations
415  // like the one we're doing here; in earlier versions (and for larger allocations) they are
416  // just wrappers around malloc.
417 #if PY_VERSION_HEX >= 0x03050000
418  nonsimple.values_and_holders = (void **) PyMem_Calloc(space, sizeof(void *));
419  if (!nonsimple.values_and_holders) throw std::bad_alloc();
420 #else
421  nonsimple.values_and_holders = (void **) PyMem_New(void *, space);
422  if (!nonsimple.values_and_holders) throw std::bad_alloc();
423  std::memset(nonsimple.values_and_holders, 0, space * sizeof(void *));
424 #endif
425  nonsimple.holder_constructed = reinterpret_cast<bool *>(&nonsimple.values_and_holders[flags_at]);
426  }
427  owned = true;
428 }

531  const detail::type_info *tinfo,
532  void *(*copy_constructor)(const void *),
533  void *(*move_constructor)(const void *),
534  const void *existing_holder = nullptr) {
535  if (!tinfo) // no type info: error will be set already
536  return handle();
537 
538  void *src = const_cast<void *>(_src);
539  if (src == nullptr)
540  return none().release();
541 
542  auto it_instances = get_internals().registered_instances.equal_range(src);
543  for (auto it_i = it_instances.first; it_i != it_instances.second; ++it_i) {
544  for (auto instance_type : detail::all_type_info(Py_TYPE(it_i->second))) {
545  if (instance_type && instance_type == tinfo)
546  return handle((PyObject *) it_i->second).inc_ref();

551  auto wrapper = reinterpret_cast<instance *>(inst.ptr());
552  wrapper->owned = false;
553  void *&valueptr = values_and_holders(wrapper).begin()->value_ptr();
554 
555  switch (policy) {
556  case return_value_policy::automatic:
557  case return_value_policy::take_ownership:
558  valueptr = src;
559  wrapper->owned = true;
560  break;
561 
562  case return_value_policy::automatic_reference:
563  case return_value_policy::reference:
564  valueptr = src;
565  wrapper->owned = false;
566  break;
567 
568  case return_value_policy::copy:

570  valueptr = copy_constructor(src);
571  else
572  throw cast_error("return_value_policy = copy, but the "
573  "object is non-copyable!");
574  wrapper->owned = true;
575  break;
576 
577  case return_value_policy::move:
578  if (move_constructor)
579  valueptr = move_constructor(src);
580  else if (copy_constructor)
581  valueptr = copy_constructor(src);
582  else

586  break;

596  }

631  // logic (without having to resort to virtual inheritance).
632  template <typename ThisT>
633  PYBIND11_NOINLINE bool load_impl(handle src, bool convert) {
634  if (!src || !typeinfo)
635  return false;
636  if (src.is_none()) {
637  // Defer accepting None to other overloads (if we aren't in convert mode):
638  if (!convert) return false;

647 
648  // Case 1: If src is an exact type match for the target type then we can reinterpret_cast
649  // the instance's value pointer to the target type:
650  if (srctype == typeinfo->type) {
651  this_.load_value(reinterpret_cast<instance *>(src.ptr())->get_value_and_holder());
652  return true;
653  }
654  // Case 2: We have a derived class
655  else if (PyType_IsSubtype(srctype, typeinfo->type)) {
656  auto &bases = all_type_info(srctype);
657  bool no_cpp_mi = typeinfo->simple_type;
658 
659  // Case 2a: the python type is a Python-inherited derived class that inherits from just
660  // one simple (no MI) pybind11 class, or is an exact match, so the C++ instance is of
661  // the right type and we can use reinterpret_cast.
662  // (This is essentially the same as case 2b, but because not using multiple inheritance
663  // is extremely common, we handle it specially to avoid the loop iterator and type
664  // pointer lookup overhead)
665  if (bases.size() == 1 && (no_cpp_mi || bases.front()->type == typeinfo->type)) {
666  this_.load_value(reinterpret_cast<instance *>(src.ptr())->get_value_and_holder());
667  return true;
668  }
669  // Case 2b: the python type inherits from multiple C++ bases. Check the bases to see if
670  // we can find an exact match (or, for a simple C++ type, an inherited match); if so, we
671  // can safely reinterpret_cast to the relevant pointer.
672  else if (bases.size() > 1) {
673  for (auto base : bases) {
674  if (no_cpp_mi ? PyType_IsSubtype(base->type, typeinfo->type) : base->type == typeinfo->type) {
675  this_.load_value(reinterpret_cast<instance *>(src.ptr())->get_value_and_holder(base));
676  return true;
677  }
678  }
679  }
680 
681  // Case 2c: C++ multiple inheritance is involved and we couldn't find an exact type match
682  // in the registered bases, above, so try implicit casting (needed for proper C++ casting
683  // when MI is involved).
684  if (this_.try_implicit_casts(src, convert))
685  return true;
686  }

688  // Perform an implicit conversion
689  if (convert) {
690  for (auto &converter : typeinfo->implicit_conversions) {
691  auto temp = reinterpret_steal<object>(converter(src.ptr(), typeinfo->type));
692  if (load_impl<ThisT>(temp, false)) {

695  }

699  }

701  }
