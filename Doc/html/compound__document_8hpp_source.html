70 struct compound_document_entry
71 {
72  void name(const std::string &new_name)
73  {
74  auto u16_name = utf8_to_utf16(new_name);
75  name_length = std::min(static_cast<std::uint16_t>(u16_name.size()), std::uint16_t(31));
76  std::copy(u16_name.begin(), u16_name.begin() + name_length, name_array.begin());
77  name_array[name_length] = 0;
78  name_length = (name_length + 1) * 2;
79  }
80 
81  std::string name() const
82  {
83  return utf16_to_utf8(std::u16string(name_array.begin(),
84  name_array.begin() + (name_length - 1) / 2));
85  }
